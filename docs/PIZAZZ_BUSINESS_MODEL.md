export async function getDistributorPrice(
  productSku: string,
  distributorId?: string,
  totalQuantity?: number

    where: { sku: productSku }
  });

  if (!distributorId) {
    // B2C ê³ ê° (ì¼ë°˜ ê°€ê²©)

    price: discountedPrice,
    discount: discountRate,
    quantity: quantity,
    type: "DISTRIBUTOR"
  };
}

// app/api/quote/generate.ts

export async function generateQuote(
  items: QuoteItem[],
  distributorId: string,
  poNumber: string
): Promise<Quote> {
  const distributor = await prisma.distributor.findUnique({
    where: { id: distributorId }
  });

  let total = 0;
  const lineItems = [];

  for (const item of items) {
    const price = await getDistributorPrice(item.sku, distributorId);
    const subtotal = price.price * item.quantity;

    lineItems.push({
      sku: item.sku,
      description: item.description,
      quantity: item.quantity,
      unit_price: price.price,
      subtotal: subtotal
    });

    total += subtotal;
  }

  const quote = await prisma.quote.create({
    data: {
      distributor_id: distributorId,
      po_number: poNumber,
      items: lineItems,
      subtotal: total,
      shipping: 200, // ê³ ì • ë°°ì†¡ë¹„
      tax: total * 0.1, // ì„¸ê¸ˆ 10%
      total: total + 200 + total * 0.1,
      quote_number: `${distributor.invoice_prefix}Q-${Date.now()}`,
      expires_at: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) // 30ì¼
    }
  });

  // ì´ë©”ì¼ë¡œ ë°œì†¡
